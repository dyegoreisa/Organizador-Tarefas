796       def route(verb, path, options={}, &block)
 
797         # Because of self.options.host
 
798         host_name(options.delete(:host)) if options.key?(:host)
 
799 
800         options.each {|option, args| send(option, *args)}
 
801 
802         pattern, keys = compile(path)
 
803         conditions, @conditions = @conditions, []
 
804 
805         define_method "#{verb} #{path}", &block
 
806         unbound_method = instance_method("#{verb} #{path}")
 
807         block =
 
808           if block.arity != 0
 
809             lambda { unbound_method.bind(self).call(*@block_params) }
 
810           else
 
811             lambda { unbound_method.bind(self).call }
 
812           end
 
813 
814         invoke_hook(:route_added, verb, path, block)

816         (routes[verb] ||= []).
 
817           push([pattern, keys, conditions, block]).last

824       def compile(path)
 
825         keys = []
 
826         if path.respond_to? :to_str
 
827           special_chars = %w{. + ( )}
 
828           pattern =
 
829             path.to_str.gsub(/((:\w+)|[\*#{special_chars.join}])/) do |match|
 
830               case match
 
831               when "*"
 
832                 keys << 'splat'
 
833                 "(.*?)"
 
834               when *special_chars
 
835                 Regexp.escape(match)
 
836               else
 
837                 keys << $2[1..-1]
 
838                 "([^/?&#]+)"
 
839               end
 
840             end
 
841           [/^#{pattern}$/, keys]
 
842         elsif path.respond_to?(:keys) && path.respond_to?(:match)
 
843           [path, path.keys]
 
844         elsif path.respond_to? :match
 
845           [path, keys]
 
846         else
 
847           raise TypeError, path
